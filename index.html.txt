    <script>
        function checkBirthday() {
            let inputDate = document.getElementById("birthdayInput").value;
            let momBirthday = "1981年3月8日"; // 这里是你妈妈的生日

            if (inputDate === momBirthday) {
                document.getElementById("message").innerHTML = "🎉 生日快乐！妈妈！";
            } else {
                document.getElementById("message").innerHTML = "这个日期好像不对哦~ 再试试！尝试使用年月日的格式哦";
            }
        }
